4   import java.util.LinkedList;
5   import java.util.Map;
6   import randomunit.MethodInvocationLog.PooledObject;

8   /**
9    * A LogStrategy implementation that keeps a history log for every object that appears either as a
10   * parameter or as a returned value. Note that the log queues are categorized by the object pools names,
11   * so if it happens that two equal objects exist in two separate object pools, two queues will exist;
12   * each object will have its own history record.
13   *
14   * @see randomunit.RandomizedTestCase
15   * @author Andreou Dimitris, andreou &lt at &gt csd dot uoc dot gr
16   */
17  public class DetailedLogStrategy implements LogStrategy {
18      private final int bufferPerObjectSize;
19      private final Map<String, Map<Object, LinkedList<MethodInvocationLog>>> logMap =
20              new HashMap<String, Map<Object, LinkedList<MethodInvocationLog>>>();
21      
22      /**
23       * Creates a DetailedLogStrategy, that will maintain a queue of log entries for every method
24       * parameter or return value. The queue will have at most the specified size.
25       *
26       * @param bufferPerObjectSize the maximum size of a single object's queue
27       */
28      public DetailedLogStrategy(int bufferPerObjectSize) {
29          if (bufferPerObjectSize < 0) {
30              throw new IllegalArgumentException("Buffer length cannot be negative");
31          }
32          this.bufferPerObjectSize = bufferPerObjectSize;
33      }
34      
35      public void appendLog(MethodInvocationLog log) {
36          for (PooledObject arg : log.getArgs()) {
37              updateLog(getObjectLog(arg), log);
38          }
39          Object ret = log.getReturnedValue();
40          if (ret != null) {
41              for (String poolName : log.getTargetPools()) {            
42                  updateLog(getObjectLog(new PooledObject(ret, poolName)), log);
43              }
44          }
45      }
46      
47      private void updateLog(LinkedList<MethodInvocationLog> logEntries, MethodInvocationLog log) {
48          logEntries.addLast(log);
49          if (logEntries.size() > bufferPerObjectSize) {
50              logEntries.removeFirst();
51          }
52      }
53      
54      private LinkedList<MethodInvocationLog> getObjectLog(PooledObject pooledObject) {
55          Map<Object, LinkedList<MethodInvocationLog>> map = logMap.get(pooledObject.getPoolName());
56          if (map == null) {
57              map = new HashMap<Object, LinkedList<MethodInvocationLog>>();
58              logMap.put(pooledObject.getPoolName(), map);
59          }
60          LinkedList<MethodInvocationLog> list = map.get(pooledObject.getObject());
61          if (list == null) {
62              list = new LinkedList<MethodInvocationLog>();
63              map.put(pooledObject.getObject(), list);
64          }
65          return list;
66      }
67      
68      public String dump() {
69          StringBuilder sb = new StringBuilder();
70          sb.append("[");
71          for (String pool : logMap.keySet()) {
72              sb.append("Pool:'").append(pool).append("'=");
73              sb.append(logMap.get(pool).toString());
74              sb.append(", ");
75          }
76          if (!logMap.isEmpty()) {
77              sb.delete(sb.length() - 2, sb.length());
78          }
79          sb.append("]");
80          
81          return sb.toString();
82      }
