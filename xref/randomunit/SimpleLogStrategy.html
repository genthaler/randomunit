16       * Creates an instance of this class, given the maximum allowable length of the internal buffer.

20       */
21      public SimpleLogStrategy(int bufferLength) {

23              throw new IllegalArgumentException("Buffer length cannot be negative");

26      }

30      }
31  
32      public void appendLog(MethodInvocationLog entry) {
